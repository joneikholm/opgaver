Hele Hello World Programmet

section .data
    msg db "Hello world!", 0ah

section .text
    global _start

_start:
    mov rax, 1      ; systemkald: write
    mov rdi, 1      ; fil-deskriptor (stdout)
    mov rsi, msg    ; adresse til teksten
    mov rdx, 13     ; antal bytes
    syscall

    mov rax, 60     ; systemkald: exit
    mov rdi, 0      ; exit-status
    syscall
C-ækvivalent:
write(1, "Hello world!\n", 13);
exit(0);
